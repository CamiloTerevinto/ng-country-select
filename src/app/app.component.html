Installation


ng add @wlucha/ng-country-select

1. Outline Appearance


<ng-country-select
  appearance="outline"
></ng-country-select>

2. Exclude countries


<ng-country-select
  [excludeCountries]="['US', 'DE', 'FR']"
></ng-country-select>

3. Include countries


<ng-country-select
  [includeCountries]="['US', 'DE', 'FR']"
></ng-country-select>

4. Set Default Country (non-FormControl)


<ng-country-select
  [defaultCountry]="presetCountry"
></ng-country-select>

5. Set Default Country (FormControl)


<ng-country-select
  [formControl]="countryControl"
></ng-country-select>

6. Set language & placeholder


<ng-country-select
  [lang]="'es'" // Set spanish as language
  [placeholder]="'Seleccionar pa√≠s'"
  [defaultCountry]="presetCountry"
></ng-country-select>

7. Show Alpha2/Alpha3 Codes


<ng-country-select
  [showCodes]="true"
></ng-country-select>

8. Hide Flag


<ng-country-select
  [showFlag]="false"
></ng-country-select>

9. Show Alpha2


<ng-country-select
  [alpha2Only]="true"
></ng-country-select>
